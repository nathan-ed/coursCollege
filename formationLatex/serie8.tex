En écrivant les principes d'équivalence pour chaque étape.
	\begin{tasks}(4)
\task $4 x=9 x$
\task $6 x+3=5 x$
\task $4 x-5=3 x+2$
\task $8 x=9 x+3$
\end{tasks}
De tête.
\begin{tasks}(4)
\task $4 x-7=10 x-7$
\task $5 x+1=5 x-1$

\task $7 x-8=-x$

\task $13 x-1=5 x-2$
\task $6 x+4=2 x+10$
\task $2-5 x=11-3 x$
